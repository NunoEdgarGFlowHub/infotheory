102  void setDataRanges(TVector<double>& mins, TVector<double>& maxs){
103  // init bins[r][d] with shifted values from mins[d] to maxs[d] (left margin of bins)
104 
105  TVector<double> offset, unitOffset;
106 
107  offset.SetBounds(1,nDims);
108  unitOffset.SetBounds(1,nDims);
109 
110  // set offset to left most shifted bins
111  for(int d=1; d<=nDims; d++){
112  binSizes[d] = (maxs[d]-mins[d])/nBins[d];
113  binSizes[d] += binSizes[d]/10.;
114  if(nReps==1){
115  offset[d] = mins[d]-(binSizes[d]/10.);
116  }
117  else{
118  offset[d] = mins[d]-(binSizes[d]/2.);
119  }
120  unitOffset[d] = binSizes[d]/nReps;
121  }
122 
123  // for each rep
124  for(int r=1; r<=nReps; r++){
125  // for each dimension
126  for(int d=1; d<=nDims; d++){
127  // for each bin
128  for(int b=1; b<=nBins[d]; b++){
129  bins[r][d][b] = (offset[d])+(binSizes[d]*(b-1));
130  //cout << bins[r][d][b] << " ";
131  }
132  //cout << endl << "binsizes " << binSizes[d] << endl;
133  }
134 
135  // update offset for next rep
136  for(int d=1; d<=nDims; d++){
137  offset[d] += unitOffset[d];
138  }
139  }
140 
141  //cout << "binSizes = " << binSizes << endl;
142  //cout << "Done setDataRanges" << endl;
